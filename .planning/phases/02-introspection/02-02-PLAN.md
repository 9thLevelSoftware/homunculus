<objective>
Implement the metrics introspection mode that analyzes episode records to compute quantitative performance signals: patch success rate, failure stage distribution, retry patterns, and trends over time.
</objective>

<context>
## Prior Work (Plan 02-01)
- `IntrospectionMode` protocol defined in `introspection/base.py`
- `IntrospectionContext` provides `store`, `config`, `cycle_number`, `window_size`
- `IntrospectionResult` dataclass in `models.py`
- `IntrospectionScheduler` calls metrics mode every cycle

## Data Source
`traces/episodes.jsonl` contains `EpisodeRecord` entries with:
- `outcome`: "accepted" | "reverted" | "blocked" | "error"
- `failure_stage`: stage where failure occurred (if any)
- `attempt_index`: retry count for this task
- `source`: "self-generated" | "user"
- `timestamp`: ISO timestamp for time windowing
- `comparison_group`: optional task grouping

## Existing Metrics Definition
`models.py:EvaluationMetrics` already defines:
- `compile_pass_rate`, `task_success_rate`, `average_retries_to_success`
- `regression_count`, `memory_usefulness_score`, `tool_misuse_rate`

The metrics mode should compute these plus additional introspection-specific metrics.

## Patterns to Follow
- Use `store.load_episodes()` to get all episodes
- Filter by timestamp for windowed analysis
- Return `IntrospectionResult` with findings, metrics dict, recommendations
</context>

class MetricsMode:
    """Introspection mode that computes quantitative performance metrics."""
    
    @property
    def name(self) -> str:
        return "metrics"
    
    def run(self, context: IntrospectionContext) -> IntrospectionResult:
        episodes = self._load_windowed_episodes(context)
        
        if not episodes:
            return IntrospectionResult(
                mode=self.name,
                timestamp=utc_now(),
                findings=[],
                summary="No episodes in analysis window",
                metrics={},
                recommendations=["Generate more episodes before metrics analysis"],
            )
        
        metrics = self._compute_metrics(episodes)
        findings = self._generate_findings(episodes, metrics)
        recommendations = self._generate_recommendations(metrics)

            mode=self.name,
            timestamp=utc_now(),

def _compute_metrics(self, episodes: list[EpisodeRecord]) -> dict[str, float]:
    total = len(episodes)
    if total == 0:
        return {}
    
    # Outcome distribution
    outcomes = Counter(ep.outcome for ep in episodes)
    success_rate = outcomes.get("accepted", 0) / total
    revert_rate = outcomes.get("reverted", 0) / total
    error_rate = outcomes.get("error", 0) / total
    blocked_rate = outcomes.get("blocked", 0) / total
    
    # Failure stage distribution
    failure_stages = Counter(
        ep.failure_stage for ep in episodes 
        if ep.failure_stage and ep.outcome != "accepted"
    )
    
    # Retry analysis
    retry_episodes = [ep for ep in episodes if ep.attempt_index > 1]
    avg_retries = (
        sum(ep.attempt_index for ep in retry_episodes) / len(retry_episodes)
        if retry_episodes else 0.0
    )
    retry_rate = len(retry_episodes) / total
    
    # Source distribution
    self_generated = sum(1 for ep in episodes if ep.source == "self-generated")
    self_generated_ratio = self_generated / total
    
    return {
        "success_rate": round(success_rate, 3),
        "revert_rate": round(revert_rate, 3),
        "error_rate": round(error_rate, 3),
        "blocked_rate": round(blocked_rate, 3),
        "avg_retries": round(avg_retries, 2),
        "retry_rate": round(retry_rate, 3),
        "self_generated_ratio": round(self_generated_ratio, 3),
        "total_episodes": float(total),
        # Failure stage percentages (of failures only)
        **{f"failure_{stage}": round(count / total, 3) 
           for stage, count in failure_stages.items()},
    }

def _generate_findings(self, episodes: list[EpisodeRecord], metrics: dict[str, float]) -> list[dict]:
    findings = []
    
    # Success rate finding
    findings.append({
        "type": "success_rate",
        "value": metrics.get("success_rate", 0),
        "severity": "info" if metrics.get("success_rate", 0) >= 0.7 else "warning",
        "detail": f"{int(metrics.get('success_rate', 0) * 100)}% of patches accepted",
    })
    
    # High error rate finding
    if metrics.get("error_rate", 0) > 0.1:
        findings.append({
            "type": "high_error_rate",
            "value": metrics.get("error_rate", 0),
            "severity": "critical",
            "detail": f"{int(metrics.get('error_rate', 0) * 100)}% of episodes ended in error",
        })
    
    # Retry pattern finding
    if metrics.get("retry_rate", 0) > 0.3:
        findings.append({
            "type": "high_retry_rate",
            "value": metrics.get("retry_rate", 0),
            "severity": "warning",
            "detail": f"{int(metrics.get('retry_rate', 0) * 100)}% of tasks required retries",
        })
    
    # Failure stage concentration
    failure_stages = {k: v for k, v in metrics.items() if k.startswith("failure_")}
    if failure_stages:
        worst_stage = max(failure_stages, key=failure_stages.get)
        findings.append({
            "type": "failure_concentration",
            "value": failure_stages[worst_stage],
            "severity": "info",
            "detail": f"Most failures occur at {worst_stage.replace('failure_', '')} stage",
        })
    
    return findings
```

def _generate_recommendations(self, metrics: dict[str, float]) -> list[str]:
    recommendations = []
    
    if metrics.get("success_rate", 1.0) < 0.5:
        recommendations.append(
            "Success rate below 50%. Consider reviewing recent failures for common patterns."
        )
    
    if metrics.get("error_rate", 0) > 0.1:
        recommendations.append(
            "High error rate detected. Check for infrastructure issues or invalid task prompts."
        )
    
    if metrics.get("retry_rate", 0) > 0.3:
        recommendations.append(
            "Many tasks require retries. Consider improving initial plan generation."
        )
    
    # Failure stage specific recommendations
    if metrics.get("failure_execute", 0) > 0.2:
        recommendations.append(
            "Execute stage failures high. Review patch application and worktree isolation."
        )
    
    if metrics.get("failure_plan", 0) > 0.2:
        recommendations.append(
            "Plan stage failures high. Teacher model may need prompt adjustments."
        )
    
    if not recommendations:
        recommendations.append("Metrics look healthy. Continue current approach.")
    
    return recommendations

def _format_summary(self, metrics: dict[str, float], episode_count: int) -> str:
    success = int(metrics.get("success_rate", 0) * 100)
    return (
        f"Analyzed {episode_count} episodes: {success}% success rate, "
        f"{int(metrics.get('retry_rate', 0) * 100)}% retry rate, "
        f"{int(metrics.get('self_generated_ratio', 0) * 100)}% self-generated"
    )
```
