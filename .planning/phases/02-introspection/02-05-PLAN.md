<objective>
Implement the comparative introspection mode that groups episodes by task (comparison_group), compares winning vs losing patches for the same task, and extracts patterns that distinguish successful approaches.
</objective>

<context>
## Prior Work (Plan 02-01)
- `IntrospectionMode` protocol defined in `introspection/base.py`
- `IntrospectionContext` provides `store`, `config`
- Episodes have `comparison_group` field for task grouping

## Episode Comparison Structure
From `EpisodeRecord`:
- `comparison_group`: Optional task identifier that groups related attempts
- `outcome`: "accepted" = winner, others = losers
- `patch`: The actual diff content
- `plan`: The approach taken
- `attempt_index`: Which attempt this was

## Analysis Goals
1. Find tasks with both winning and losing attempts
2. Compare patch content between winners and losers
3. Extract what made winners succeed (size, approach, etc.)
4. Identify common failure patterns in losers

```python
from __future__ import annotations
from collections import defaultdict
from typing import TYPE_CHECKING

from ..models import EpisodeRecord, IntrospectionResult, utc_now

if TYPE_CHECKING:
    from .base import IntrospectionContext


class ComparativeMode:
    """Introspection mode that compares winning vs losing patches."""
    
    @property
    def name(self) -> str:
        return "comparative"
    
    def run(self, context: IntrospectionContext) -> IntrospectionResult:
        # Load and group episodes
        episodes = context.store.load_episodes()
        groups = self._group_by_comparison(episodes)
        
        # Filter to groups with both winners and losers
        comparable = {
            group_id: eps for group_id, eps in groups.items()
            if self._has_comparison_pair(eps)
        }
        
        if not comparable:
            return IntrospectionResult(
                mode=self.name,
                timestamp=utc_now(),
                findings=[],
                summary="No comparable episode pairs found",
                metrics={"comparable_groups": 0.0},
                recommendations=[
                    "Use comparison_group in TaskRequest to enable comparative analysis"
                ],
            )
        
        # Analyze each comparable group
        findings = []
        all_patterns = []
        
        for group_id, eps in comparable.items():
            analysis = self._analyze_group(group_id, eps)
            findings.extend(analysis["findings"])
            all_patterns.extend(analysis["patterns"])
        
        # Aggregate patterns across groups
        aggregated = self._aggregate_patterns(all_patterns)
        
        return IntrospectionResult(
            mode=self.name,
            timestamp=utc_now(),
            findings=findings + aggregated["findings"],
            summary=self._format_summary(comparable, aggregated),
            metrics={
                "comparable_groups": float(len(comparable)),
                "total_comparisons": float(sum(len(eps) for eps in comparable.values())),
                **aggregated["metrics"],
            },
            recommendations=aggregated["recommendations"],
        )
    
    def _group_by_comparison(self, episodes: list[EpisodeRecord]) -> dict[str, list[EpisodeRecord]]:
        """Group episodes by comparison_group."""
        groups: dict[str, list[EpisodeRecord]] = defaultdict(list)
        for ep in episodes:
            if ep.comparison_group:
                groups[ep.comparison_group].append(ep)
        return dict(groups)
    
    def _has_comparison_pair(self, episodes: list[EpisodeRecord]) -> bool:
        """Check if group has both winners and losers."""
        has_winner = any(ep.outcome == "accepted" for ep in episodes)
        has_loser = any(ep.outcome != "accepted" for ep in episodes)
        return has_winner and has_loser
    
    def _analyze_group(self, group_id: str, episodes: list[EpisodeRecord]) -> dict:
```

```python
def _analyze_group(self, group_id: str, episodes: list[EpisodeRecord]) -> dict:
    """Analyze a single comparison group for patterns."""
    winners = [ep for ep in episodes if ep.outcome == "accepted"]
    losers = [ep for ep in episodes if ep.outcome != "accepted"]
    
    findings = []
    patterns = []
    
    # Basic stats
    findings.append({
        "type": "group_stats",
        "group": group_id,
        "severity": "info",
        "detail": f"Group '{group_id[:20]}': {len(winners)} winners, {len(losers)} losers",
        "value": len(winners) / (len(winners) + len(losers)),
    })
    
    # Analyze winning patches
    for winner in winners:
        win_analysis = self._analyze_patch(winner)
        patterns.append({
            "type": "winner",
            "group": group_id,
            **win_analysis,
        })
    
    # Analyze losing patches and compare
    for loser in losers:
        lose_analysis = self._analyze_patch(loser)
        patterns.append({
            "type": "loser",
            "group": group_id,
            "failure_stage": loser.failure_stage,
            **lose_analysis,
        })
    
    # Compare winner vs loser characteristics
    if winners and losers:
        comparison = self._compare_patches(winners[0], losers[0])
        if comparison:
            findings.append({
                "type": "comparison",
                "group": group_id,
                "severity": "info",
                "detail": comparison["insight"],
                "value": comparison.get("metric", 0),
            })
    
    return {"findings": findings, "patterns": patterns}

def _analyze_patch(self, episode: EpisodeRecord) -> dict:
    """Extract characteristics from a patch."""
    patch = episode.patch or ""
    
    # Patch size metrics
    lines = patch.count("\n")
    additions = patch.count("\n+") - patch.count("\n+++")  # Exclude header
    deletions = patch.count("\n-") - patch.count("\n---")
    
    # Plan complexity
    plan_steps = len(episode.plan) if episode.plan else 0
    
    return {
        "patch_lines": lines,
        "additions": max(0, additions),
        "deletions": max(0, deletions),
        "plan_steps": plan_steps,
        "attempt": episode.attempt_index,
    }

def _compare_patches(self, winner: EpisodeRecord, loser: EpisodeRecord) -> dict | None:
    """Compare characteristics of winner vs loser."""
    win_analysis = self._analyze_patch(winner)
    lose_analysis = self._analyze_patch(loser)
    
    insights = []
    
    # Size comparison
    if win_analysis["patch_lines"] < lose_analysis["patch_lines"] * 0.7:
        insights.append("Winner patch was significantly smaller")
    elif win_analysis["patch_lines"] > lose_analysis["patch_lines"] * 1.3:
        insights.append("Winner patch was larger (more thorough)")
    
    # Plan comparison
    if win_analysis["plan_steps"] < lose_analysis["plan_steps"]:
        insights.append("Winner had simpler plan")
    elif win_analysis["plan_steps"] > lose_analysis["plan_steps"]:
        insights.append("Winner had more detailed plan")
    
    # Attempt comparison
    if winner.attempt_index < loser.attempt_index:
        insights.append("Winner succeeded on earlier attempt")
    
    if not insights:
        return None
    
    return {
        "insight": "; ".join(insights),
        "metric": win_analysis["patch_lines"] / max(lose_analysis["patch_lines"], 1),
    }
```

```python
def _aggregate_patterns(self, patterns: list[dict]) -> dict:
    """Aggregate patterns across all groups to find trends."""
    findings = []
    recommendations = []
    
    # Separate winners and losers
    winners = [p for p in patterns if p["type"] == "winner"]
    losers = [p for p in patterns if p["type"] == "loser"]
    
    if not winners or not losers:
        return {"findings": [], "metrics": {}, "recommendations": []}
    
    # Average characteristics
    avg_winner_lines = sum(p["patch_lines"] for p in winners) / len(winners)
    avg_loser_lines = sum(p["patch_lines"] for p in losers) / len(losers)
    
    avg_winner_steps = sum(p["plan_steps"] for p in winners) / len(winners)
    avg_loser_steps = sum(p["plan_steps"] for p in losers) / len(losers)
    
    # Size insight
    if avg_winner_lines < avg_loser_lines * 0.8:
        findings.append({
            "type": "size_pattern",
            "severity": "info",
            "detail": f"Winning patches average {avg_winner_lines:.0f} lines vs {avg_loser_lines:.0f} for losers",
            "value": avg_winner_lines / max(avg_loser_lines, 1),
        })
        recommendations.append("Prefer smaller, focused patches over large changes")
    
    # Plan insight
    if avg_winner_steps < avg_loser_steps * 0.8:
        findings.append({
            "type": "plan_pattern",
            "severity": "info",
            "detail": f"Winners average {avg_winner_steps:.1f} plan steps vs {avg_loser_steps:.1f} for losers",
            "value": avg_winner_steps / max(avg_loser_steps, 1),
        })
        recommendations.append("Simpler plans tend to succeed more often")
    
    # Failure stage distribution in losers
    from collections import Counter
    failure_stages = Counter(p.get("failure_stage") for p in losers if p.get("failure_stage"))
    if failure_stages:
        most_common = failure_stages.most_common(1)[0]
        findings.append({
            "type": "failure_stage_pattern",
            "severity": "warning",
            "detail": f"Most failures occur at '{most_common[0]}' stage ({most_common[1]} times)",
            "value": most_common[1] / len(losers),
        })
    
    return {
        "findings": findings,
        "metrics": {
            "avg_winner_lines": round(avg_winner_lines, 1),
            "avg_loser_lines": round(avg_loser_lines, 1),
            "avg_winner_steps": round(avg_winner_steps, 1),
            "avg_loser_steps": round(avg_loser_steps, 1),
            "winner_count": float(len(winners)),
            "loser_count": float(len(losers)),
        },
        "recommendations": recommendations,
    }

def _format_summary(self, comparable: dict, aggregated: dict) -> str:
    """Format human-readable summary."""
    metrics = aggregated.get("metrics", {})
    winners = int(metrics.get("winner_count", 0))
    losers = int(metrics.get("loser_count", 0))
    groups = len(comparable)
    
    return (
        f"Compared {groups} task groups: {winners} winners vs {losers} losers. "
        f"Avg winning patch: {metrics.get('avg_winner_lines', 0):.0f} lines."
    )
```
