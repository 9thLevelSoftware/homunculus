## Teacher Client
`homunculus/orchestrator/teacher.py:OpenAICompatibleTeacher`:
- **CRITICAL**: Actual signature is `generate(task: TaskRequest, memories: list[MemoryRecord], student_hint: str | None) -> TeacherResponse`
- NOT a simple `generate(prompt: str)` - must construct a TaskRequest
- Teacher expects JSON output with `plan`, `candidate_patch`, `rationale` keys
- Existing `_extract_json()` already handles markdown-wrapped JSON

For critique analysis, construct a TaskRequest:
```python

)

```

## Episode Data for Review
From `EpisodeRecord`:
- `prompt`: The task description
- `plan`: Steps the teacher planned
- `outcome`: Result (accepted/reverted/error)
- `failure_stage`: Where it failed
- `error_message`: Error details
- `patch`: The generated patch

## Cost Considerations
- Critique mode calls the teacher API, which costs tokens
- `IntrospectionSettings.critique_enabled` can disable it
- Should batch episodes into a single analysis call
- Limit input size (truncate patches, summarize)

from __future__ import annotations
import json
from typing import TYPE_CHECKING

from ..models import EpisodeRecord, IntrospectionResult, TaskRequest, utc_now
from ..orchestrator.teacher import OpenAICompatibleTeacher

if TYPE_CHECKING:
    from .base import IntrospectionContext


class CritiqueMode:
    """Introspection mode that uses LLM to analyze episode patterns.
    
    CRITICAL: Uses OpenAICompatibleTeacher which requires TaskRequest objects,
    not raw prompts. We construct a TaskRequest where the prompt contains
    the episode summaries for analysis.
    """
    
    ANALYSIS_PROMPT = '''You are analyzing a coding agent's recent performance.

Review these episode summaries and identify patterns:

{episode_summaries}

Provide your analysis in JSON format:
{{
    "patterns": [
        {{
            "pattern": "description of observed pattern",
            "frequency": "how often it occurs",
            "severity": "low|medium|high",
            "examples": ["episode_id1", "episode_id2"]
        }}
    ],
    "weaknesses": [
        {{
            "area": "specific weakness area",
            "description": "detailed description",
            "impact": "how this affects performance",
            "recommendation": "specific actionable fix"
        }}
    ],
    "strengths": [
        "area where agent performs well"
    ],
    "summary": "one paragraph overall assessment"
}}

Focus on actionable insights. Be specific about failure patterns.'''
    
    @property
    def name(self) -> str:
        return "critique"
    
    def __init__(self, teacher: OpenAICompatibleTeacher | None = None) -> None:
        self._teacher = teacher
    
    def run(self, context: IntrospectionContext) -> IntrospectionResult:
        # Check if critique is enabled
        if not context.config.introspection.critique_enabled:
            return IntrospectionResult(
                mode=self.name,
                timestamp=utc_now(),
                findings=[],
                summary="Critique mode disabled in config",
                metrics={},
                recommendations=[],
            )
        
        # Load recent episodes
        episodes = self._load_recent_episodes(context)
        if len(episodes) < 3:
            return IntrospectionResult(
                mode=self.name,
                timestamp=utc_now(),
                findings=[],
                summary=f"Need at least 3 episodes for critique (have {len(episodes)})",
                metrics={},
                recommendations=["Generate more episodes before running critique"],
            )
        
        # Build teacher client if not injected
        # CRITICAL: Use OpenAICompatibleTeacher with TeacherSettings, not raw config
        teacher = self._teacher or OpenAICompatibleTeacher(context.config.teacher)
        
        # Generate analysis (pass cycle_number for task_id)
        analysis = self._analyze_episodes(episodes, teacher, context.cycle_number)
        
        return self._build_result(analysis, len(episodes))
    
    def _load_recent_episodes(self, context: IntrospectionContext) -> list[EpisodeRecord]:
        """Load recent episodes for analysis."""
        all_episodes = context.store.load_episodes()
        return all_episodes[-context.window_size:]

def _summarize_episode(self, ep: EpisodeRecord) -> str:
    """Create concise summary of an episode for LLM analysis."""
    outcome_emoji = {
        "accepted": "OK",
        "reverted": "FAIL",
        "error": "ERR",
        "blocked": "BLOCK",
    }.get(ep.outcome, "?")
    
    summary_parts = [
        f"[{ep.episode_id[:8]}] {outcome_emoji}",
        f"Task: {ep.prompt[:100]}...",
    ]
    
    if ep.plan:
        summary_parts.append(f"Plan steps: {len(ep.plan)}")
    
    if ep.outcome != "accepted":
        if ep.failure_stage:
            summary_parts.append(f"Failed at: {ep.failure_stage}")
        if ep.error_message:
            summary_parts.append(f"Error: {ep.error_message[:100]}")
    
    # Include truncated patch for pattern analysis
    if ep.patch:
        patch_preview = ep.patch[:200] + "..." if len(ep.patch) > 200 else ep.patch
        summary_parts.append(f"Patch preview:\n{patch_preview}")
    
    return "\n".join(summary_parts)

def _analyze_episodes(self, episodes: list[EpisodeRecord], teacher: OpenAICompatibleTeacher, cycle_number: int) -> dict:
    """Send episodes to teacher for pattern analysis.
    
    CRITICAL: OpenAICompatibleTeacher.generate() requires TaskRequest, not raw prompt.
    We construct a TaskRequest where the prompt contains the analysis request.
    """
    # Summarize episodes (limit to prevent token overflow)
    max_episodes = 20  # Analyze at most 20 recent episodes
    recent = episodes[-max_episodes:]
    
    summaries = []
    for ep in recent:
        summaries.append(self._summarize_episode(ep))
    
    episode_text = "\n\n---\n\n".join(summaries)
    prompt = self.ANALYSIS_PROMPT.format(episode_summaries=episode_text)
    
    # Construct TaskRequest for teacher API
    task = TaskRequest(
        task_id=f"introspection-critique-{cycle_number}",
        workspace="self",  # Analyzing self
        prompt=prompt,
        metadata={"introspection_mode": "critique"},
    )
    
    try:
        # CRITICAL: Pass TaskRequest, empty memories, no hint
        response = teacher.generate(task, memories=[], student_hint=None)
        
        # Parse JSON from response - teacher returns structured TeacherResponse
        # The raw dict contains the full API response
        raw_text = ""
        if response.raw:
            # Extract from OpenAI-style response
            choices = response.raw.get("choices", [])
            if choices:
                raw_text = choices[0].get("message", {}).get("content", "")
        
        # Fallback to rationale or plan if content not found
        if not raw_text:
            raw_text = response.rationale or str(response.plan)
        
        # Try to extract JSON from response
        import re
        json_match = re.search(r'\{[\s\S]*\}', raw_text)
        if json_match:
            return json.loads(json_match.group())
        
        # Fallback: return raw as summary
        return {
            "patterns": [],
            "weaknesses": [],
            "strengths": [],
            "summary": raw_text[:500],
        }
        
    except Exception as e:
        return {
            "patterns": [],
            "weaknesses": [],
            "strengths": [],
            "summary": f"Analysis failed: {str(e)}",
            "error": str(e),
        }
```

def _build_result(self, analysis: dict, episode_count: int) -> IntrospectionResult:
    """Convert teacher analysis to structured IntrospectionResult."""
    findings = []
    recommendations = []
    metrics = {}
    
    # Process patterns
    patterns = analysis.get("patterns", [])
    for pattern in patterns:
        severity = pattern.get("severity", "medium")
        findings.append({
            "type": "pattern",
            "value": pattern.get("frequency", "unknown"),
            "severity": "warning" if severity in ["medium", "high"] else "info",
            "detail": pattern.get("pattern", ""),
            "examples": pattern.get("examples", []),
        })
    
    # Process weaknesses
    weaknesses = analysis.get("weaknesses", [])
    for weakness in weaknesses:
        findings.append({
            "type": "weakness",
            "value": weakness.get("area", "unknown"),
            "severity": "warning",
            "detail": weakness.get("description", ""),
            "impact": weakness.get("impact", ""),
        })
        
        rec = weakness.get("recommendation")
        if rec:
            recommendations.append(rec)
    
    # Process strengths (as positive findings)
    strengths = analysis.get("strengths", [])
    for strength in strengths:
        findings.append({
            "type": "strength",
            "value": strength,
            "severity": "info",
            "detail": f"Identified strength: {strength}",
        })
    
    # Metrics
    metrics["patterns_found"] = float(len(patterns))
    metrics["weaknesses_found"] = float(len(weaknesses))
    metrics["strengths_found"] = float(len(strengths))
    metrics["episodes_analyzed"] = float(episode_count)
    
    # Summary
    summary = analysis.get("summary", "")
    if not summary:
        summary = (
            f"Analyzed {episode_count} episodes. "
            f"Found {len(patterns)} patterns, {len(weaknesses)} weaknesses, "
            f"{len(strengths)} strengths."
        )
    
    # Handle analysis errors
    if "error" in analysis:
        findings.append({
            "type": "analysis_error",
            "value": "error",
            "severity": "warning",
            "detail": analysis["error"],
        })
    
    return IntrospectionResult(
        mode=self.name,
        timestamp=utc_now(),
        findings=findings,
        summary=summary[:500],  # Truncate long summaries
        metrics=metrics,
        recommendations=recommendations,
    )
```
