## Target Repository
The introspection runs on the Homunculus codebase itself (self-targeting).
- Source: `homunculus/` directory
- Tests: `tests/` directory
- Coverage config: May need to create `.coveragerc` or use `pyproject.toml`

## Subprocess Patterns
- See `task_runner/runner.py` for subprocess execution patterns
- Use `subprocess.run()` with capture and timeout
- Graceful degradation: if tool not installed, skip with warning
- **CRITICAL**: Use `sys.executable` instead of `"python"` for cross-platform compatibility

## External Tools
- `pytest-cov`: Test coverage (`pip install pytest-cov`)
- `coverage`: JSON report generation (installed with pytest-cov)
- `vulture`: Dead code detection (optional, `pip install vulture`)
- Built-in: Pure Python for TODO/FIXME scanning (no shell grep)

## Coverage JSON Mechanism
**CRITICAL**: pytest-cov does NOT support `--cov-report=json:path` directly.
Instead, use two-step approach:
1. `pytest --cov=module --cov-report=` (generates .coverage file)
2. `coverage json -o output.json` (converts to JSON)

```python
def _run_coverage(self, context: IntrospectionContext) -> dict | None:
    """Run pytest-cov and parse JSON results.
    
    CRITICAL: Uses sys.executable for cross-platform compatibility.
    Uses unique temp file for coverage output to avoid collisions.
    Runs `coverage json` separately since pytest-cov doesn't support direct JSON output.
    """
    import sys
    import tempfile
    
    root = context.config.paths.root
    
    # Check if pytest and coverage are available
    try:
        result = subprocess.run(
            [sys.executable, "-m", "pytest", "--version"],
            capture_output=True,
            timeout=10,
        )
        if result.returncode != 0:
            return None
        
        # Also check coverage is installed
        result = subprocess.run(
            [sys.executable, "-m", "coverage", "--version"],
            capture_output=True,
            timeout=10,
        )
        if result.returncode != 0:
            return {
                "findings": [],
                "metrics": {},
                "recommendations": ["Install coverage: pip install coverage pytest-cov"],
            }
    except (subprocess.TimeoutExpired, FileNotFoundError):
        return None
    
    # Use unique temp file to avoid collisions with concurrent runs
    with tempfile.NamedTemporaryFile(suffix=".json", delete=False) as tmp:
        cov_json = Path(tmp.name)
    
    try:
        # Step 1: Run pytest with coverage (writes .coverage file)
        result = subprocess.run(
            [
                sys.executable, "-m", "pytest",
                "--cov=homunculus",
                "--cov-report=",  # Suppress terminal output
                "--no-header", "-q",
            ],
            capture_output=True,
            timeout=300,
            cwd=root,
        )
        
        # Step 2: Generate JSON report from .coverage file
        result = subprocess.run(
            [sys.executable, "-m", "coverage", "json", "-o", str(cov_json)],
            capture_output=True,
            timeout=60,
            cwd=root,
        )
        if result.returncode != 0:
            return {
                "findings": [],
                "metrics": {},
                "recommendations": ["Coverage JSON export failed. Check .coverage file exists."],
            }
            
    except subprocess.TimeoutExpired:
        return {
            "findings": [{"type": "coverage_timeout", "severity": "warning"}],
            "metrics": {},
            "recommendations": ["Coverage run timed out. Consider optimizing tests."],
        }
    
    # Parse JSON report if it exists
    if not cov_json.exists():
        return {
            "findings": [],
            "metrics": {},
            "recommendations": ["Install pytest-cov: pip install pytest-cov"],
        }
    
    try:
        cov_data = json.loads(cov_json.read_text(encoding="utf-8"))
        total_pct = cov_data.get("totals", {}).get("percent_covered", 0)
        
        # Find low-coverage files
        low_coverage_files = []
        for filename, data in cov_data.get("files", {}).items():
            file_pct = data.get("summary", {}).get("percent_covered", 0)
            if file_pct < 50:
                low_coverage_files.append({
                    "file": filename,
                    "coverage": round(file_pct, 1),
                })
        
        findings = [{
            "type": "total_coverage",
            "value": round(total_pct, 1),
            "severity": "info" if total_pct >= 70 else "warning",
            "detail": f"Overall test coverage: {total_pct:.1f}%",
        }]
        
        if low_coverage_files:
            findings.append({
                "type": "low_coverage_files",
                "value": len(low_coverage_files),
                "severity": "warning",
                "detail": f"{len(low_coverage_files)} files have <50% coverage",
                "files": low_coverage_files[:10],  # Top 10
            })
        
        recommendations = []
        if total_pct < 70:
            recommendations.append(f"Coverage at {total_pct:.1f}%. Target 70%+.")
        
        return {
            "findings": findings,
            "metrics": {"coverage_percent": round(total_pct, 1)},
            "recommendations": recommendations,
        }
        
    except (json.JSONDecodeError, KeyError):
        return {"findings": [], "metrics": {}, "recommendations": []}
    finally:
        # Cleanup
        if cov_json.exists():
            cov_json.unlink()

def _scan_todos(self, context: IntrospectionContext) -> dict:
    """Scan source files for TODO/FIXME comments."""
    root = context.config.paths.root
    source_dir = root / "homunculus"
    
    todo_pattern = re.compile(r"#\s*(TODO|FIXME|XXX|HACK)[\s:]+(.+)", re.IGNORECASE)
    todos = []
    
    for py_file in source_dir.rglob("*.py"):
        try:
            content = py_file.read_text(encoding="utf-8")
            for lineno, line in enumerate(content.splitlines(), 1):
                match = todo_pattern.search(line)
                if match:
                    todos.append({
                        "file": str(py_file.relative_to(root)),
                        "line": lineno,
                        "type": match.group(1).upper(),
                        "text": match.group(2).strip()[:100],
                    })
        except (OSError, UnicodeDecodeError):
            continue
    
    findings = []
    if todos:
        findings.append({
            "type": "todo_comments",
            "value": len(todos),
            "severity": "info" if len(todos) < 10 else "warning",
            "detail": f"Found {len(todos)} TODO/FIXME comments",
            "items": todos[:20],  # Top 20
        })
    
    return {
        "findings": findings,
        "metrics": {
            "todo_count": float(len(todos)),
            "fixme_count": float(sum(1 for t in todos if t["type"] == "FIXME")),
        },
    }
```

```python
def _find_test_gaps(self, context: IntrospectionContext) -> dict:
    """Find source modules without corresponding test files."""
    root = context.config.paths.root
    source_dir = root / "homunculus"
    test_dir = root / "tests"
    
    # Get all source modules (excluding __init__.py, __main__.py)
    source_modules = set()
    for py_file in source_dir.rglob("*.py"):
        name = py_file.stem
        if name.startswith("_"):
            continue
        source_modules.add(name)
    
    # Get all tested modules (from test_*.py files)
    tested_modules = set()
    for test_file in test_dir.glob("test_*.py"):
        # test_foo.py -> foo
        module_name = test_file.stem.replace("test_", "", 1)
        tested_modules.add(module_name)
    
    # Find gaps
    untested = source_modules - tested_modules
    
    findings = []
    recommendations = []
    
    if untested:
        findings.append({
            "type": "untested_modules",
            "value": len(untested),
            "severity": "warning" if len(untested) > 3 else "info",
            "detail": f"{len(untested)} modules lack dedicated test files",
            "modules": sorted(untested)[:10],
        })
        
        if len(untested) > 5:
            recommendations.append(
                f"Add tests for: {', '.join(sorted(untested)[:3])}..."
            )
    
    return {
        "findings": findings,
        "recommendations": recommendations,
    }

def _format_summary(self, metrics: dict[str, float]) -> str:
    coverage = metrics.get("coverage_percent", 0)
    todos = int(metrics.get("todo_count", 0))
    
    parts = []
    if coverage > 0:
        parts.append(f"{coverage:.1f}% coverage")
    if todos > 0:
        parts.append(f"{todos} TODOs")
    
    return f"Code quality: {', '.join(parts)}" if parts else "Code quality analysis complete"
```
