## Task 3: Implement scheduler.py

**Action**: Create `homunculus/introspection/scheduler.py`:

```python
from __future__ import annotations
from dataclasses import dataclass
from typing import TYPE_CHECKING

if TYPE_CHECKING:
    from ..config import HomunculusConfig

@dataclass
class ScheduledModes:
    """Which modes should run this cycle."""
    metrics: bool = False
    critique: bool = False
    coverage: bool = False
    comparative: bool = False
    
    def any_scheduled(self) -> bool:
        return any([self.metrics, self.critique, self.coverage, self.comparative])
    
    def scheduled_names(self) -> list[str]:
        names = []
        if self.metrics: names.append("metrics")
        if self.critique: names.append("critique")
        if self.coverage: names.append("coverage")
        if self.comparative: names.append("comparative")
        return names


class IntrospectionScheduler:
    """Determines which introspection modes run each cycle."""
    
    def __init__(self, config: HomunculusConfig) -> None:
        self.config = config
        self.settings = config.introspection
    
    def get_scheduled_modes(self, cycle_number: int) -> ScheduledModes:
        """Return which modes should run for the given cycle number.
        
        CRITICAL: cycle_number must be >= 1. Cycle 0 is skipped to avoid
        the modulo edge case where 0 % n == 0 for all n, which would
        trigger all modes simultaneously on the first cycle.
        """
        if not self.settings.enabled:
            return ScheduledModes()
        
        # Skip cycle 0 to avoid modulo edge case (0 % n == 0 for all n)
        if cycle_number < 1:
            return ScheduledModes()
        
        return ScheduledModes(
            metrics=(cycle_number % self.settings.metrics_interval == 0),
            critique=(
                self.settings.critique_enabled and 
                cycle_number % self.settings.critique_interval == 0
            ),
            coverage=(cycle_number % self.settings.coverage_interval == 0),
            comparative=(cycle_number % self.settings.comparative_interval == 0),
        )
```

## Task 4: Add result persistence to ArtifactStore

**Action**: Add method to `homunculus/storage.py` for storing introspection results:

```python
def append_introspection_result(self, result: "IntrospectionResult") -> None:
    """Append introspection result to traces/introspection.jsonl."""
    self.append_jsonl(
        self.traces_dir / "introspection.jsonl",
        result.to_dict()
    )

def load_introspection_results(self, mode: str | None = None) -> list["IntrospectionResult"]:
    """Load introspection results, optionally filtered by mode."""
    from .models import IntrospectionResult
    results = [
        IntrospectionResult.from_dict(item) 
        for item in self.load_jsonl(self.traces_dir / "introspection.jsonl")
    ]
    if mode:
        results = [r for r in results if r.mode == mode]
    return results
```
