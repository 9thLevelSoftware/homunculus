## Task 2: Implement priority scoring with edge case handling
**Action**: Add private scoring methods with validation and bounds checking.

```python
def __post_init__(self) -> None:
    """Validate weights sum to approximately 1.0."""
    total = self.weights.alignment + self.weights.complexity + self.weights.freshness
    if not (0.99 <= total <= 1.01):
        raise ValueError(f"Priority weights must sum to 1.0, got {total}")

def _score_alignment(
    self,
    task: GeneratedTask,
    introspection_results: list[IntrospectionResult] | None
) -> float:
    """Score how well task aligns with current weaknesses (0.0-1.0)."""
    # Introspection-generated tasks have high alignment by definition
    if task.source == "introspection":
        return 1.0
    
    # User suggestions use their existing priority as alignment proxy
    if task.source == "user":
        return min(1.0, max(0.0, task.priority))  # Clamp to [0,1]
    
    return 0.5  # Default for other/unknown sources

def _score_complexity(self, task: GeneratedTask) -> float:
    """Estimate inverse complexity (0.0=complex, 1.0=simple)."""
    # Heuristic: shorter prompts tend to be simpler tasks
    prompt_length = len(task.prompt) if task.prompt else 0
    if prompt_length < 200:
        return 1.0  # Very simple
    elif prompt_length < 500:
        return 0.7  # Moderate
    elif prompt_length < 1000:
        return 0.4  # Complex
    return 0.2  # Very complex

def _score_freshness(self, task: GeneratedTask) -> float:
    """Score task freshness (0.0=stale, 1.0=fresh)."""
    from datetime import datetime, timezone
    
    if not task.created_at:
        return 0.5  # Default for missing timestamp
    
    try:
        created = datetime.fromisoformat(task.created_at.replace("Z", "+00:00"))
        now = datetime.now(timezone.utc)
        age_hours = (now - created).total_seconds() / 3600
        
        # Decay: 100% at 0h, ~50% at 24h, ~10% at 72h
        decay_rate = 0.03  # Per hour
        return max(0.1, 1.0 - (age_hours * decay_rate))
    except (ValueError, TypeError):
        return 0.5  # Default for invalid timestamp

def _calculate_final_priority(
    self,
    task: GeneratedTask,
    introspection_results: list[IntrospectionResult] | None
) -> float:
    """Calculate weighted priority score, clamped to [0.0, 1.0]."""
    alignment = self._score_alignment(task, introspection_results)
    complexity = self._score_complexity(task)
    freshness = self._score_freshness(task)
    
    raw_score = (
        self.weights.alignment * alignment +
        self.weights.complexity * complexity +
        self.weights.freshness * freshness
    )
    return min(1.0, max(0.0, raw_score))  # Clamp to [0, 1]
```

def _deduplicate(self, tasks: list[GeneratedTask]) -> list[GeneratedTask]:
    """Remove duplicate tasks, keeping highest priority."""
    # Key: normalized prompt prefix (first 100 chars, lowercased)
    seen: dict[str, GeneratedTask] = {}
    
    for task in tasks:
        key = task.prompt[:100].lower().strip()
        if key not in seen or task.priority > seen[key].priority:
            seen[key] = task
    
    return list(seen.values())
```

def get_pending_tasks(self) -> list[GeneratedTask]:
    """Get all pending tasks from suggestions and introspection."""
    introspection_results = self._get_recent_introspection()
    
    # Generate tasks from introspection (if generator available)
    generated = []
    if self.task_generator and introspection_results:
        generated = self.task_generator.generate_from_introspection(
            introspection_results,
            max_tasks=3
        )
    
    # Read suggestions with resonance boost (fallback to regular read)
    if introspection_results:
        suggestions = self.suggestion_reader.read_pending_with_resonance(
            introspection_results
        )
    else:
        suggestions = self.suggestion_reader.read_pending()
    
    # Combine and prioritize
    all_tasks = generated + suggestions
    return self.prioritizer.prioritize(all_tasks, introspection_results)
```
